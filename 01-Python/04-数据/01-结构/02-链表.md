# 1. 链表

由于顺序表, 需要预先知道数据大小来申请连续的内存空间, 扩充时需要数据的迁移. 所以用起来不是很灵活.

链表结构可以充分利用计算机空间, 实现内存的动态管理

链表(linked list)是一种基础的数据结构, 是线性表.

## 1.1 单向链表

### 1.1.1 图解

单向链表也叫单链表. 包含信息域和连接域.

![单向链表图示](image/02-%E9%93%BE%E8%A1%A8/%E5%8D%95%E9%93%BE%E8%A1%A8%E7%9A%84%E8%8A%82%E7%82%B9%E5%92%8C%E5%8D%95%E9%93%BE%E8%A1%A8.png)

### 1.1.2 实现

需要实现的操作:

* is_empty(): 非空校验判断
* length(): 长度
* travel(): 遍历整个链表
* add(item): 头部增加
* append(item): 尾部增加
* insert(pos, item): 选择插入
* remove(item): 删除
* search(item): 查找

```python
class Node(object):
    """单链表的节点"""

    def __init__(self, item):
        self.item = item
        self.next = None


class SingeLinkList(object):
    """单链表"""

    def __init__(self):
        self._head = None

    def is_empty(self):
        """是否为空"""
        return self._head is None

    def length(self):
        """链表长度"""
        next_node = self._head
        count = 0
        while next_node is not None:
            count += 1
            next_node = next_node.next
        return count

    def travel(self):
        """遍历整个链表"""
        next_node = self._head
        while next_node is not None:
            yield next_node.item
            next_node = next_node.next

    def add(self, item):
        """在链表头部增加节点"""
        node = Node(item)
        node.next = self._head
        self._head = node

    def append(self, item):
        """在链表尾部增加节点"""
        node = Node(item)
        if self.is_empty():
            self._head = node
        else:
            next_node = self._head
            while next_node.next is not None:
                next_node = next_node.next
            next_node.next = node

    def insert(self, pos, item):
        """在链表指定位置增加节点"""
        if pos <= 0:
            self.add(item)
        elif pos > self.length() - 1:
            self.append(item)
        else:
            node = Node(item)
            count = 0
            next_node = self._head
            while next_node.next is not None:
                count += 1
                next_node = next_node.next
            node.next = next_node.next
            next_node.next = node

    def remove(self, item):
        """移除指定位置的节点"""
        next_node = self._head
        pre_node = self._head
        while next_node.next is not None:
            if next_node.item == item:
                if pre_node:
                    pre_node.next = next_node.next
                else:
                    self._head = next_node.next
                return
            pre_node = next_node
            next_node = next_node.next
        raise IndexError(f"not found {item}")

    def search(self, item):
        """查找指定节点"""
        next_node = self._head
        while next_node is not None:
            if next_node.item == item:
                return True
            next_node = next_node.next
        return False

    def __str__(self):
        next_node = self._head
        print_str = ""
        while next_node is not None:
            print_str += f"{next_node.item} --> "
            next_node = next_node.next
        print_str += "None"
        return print_str


if __name__ == '__main__':
    sll = SingeLinkList()
    sll.append(1)
    sll.insert(0, 2)
    sll.insert(1, 3)
    sll.insert(1, 4)
    print(sll.length())
    print(sll)
    sll.remove(3)
    print(sll.length())
    print(sll)
    sll.add(5)
    print(sll)
    print(sll.search(5))
    for i in sll.travel():
        print(i)

```

## 1.2 双向链表

### 1.2.1 图解

每个节点有两个链接：一个指向前一个节点，当此节点为第一个节点时，指向空值；而另一个指向下一个节点，当此节点为最后一个节点时，指向空值。

![双向链表](image/02-%E9%93%BE%E8%A1%A8/%E5%8F%8C%E5%90%91%E9%93%BE%E8%A1%A8.png)

![双向链表指定位置插入元素](image/02-%E9%93%BE%E8%A1%A8/%E5%8F%8C%E5%90%91%E9%93%BE%E8%A1%A8%E6%8C%87%E5%AE%9A%E4%BD%8D%E7%BD%AE%E6%8F%92%E5%85%A5%E5%85%83%E7%B4%A0.png)

### 1.2.2 实现

```python
class Node(object):
    """链表节点"""

    def __init__(self, item):
        self.previous = None
        self.next = None
        self.item = item

    def __str__(self):
        return f"<previous={self.previous}, next={self.next}, item={self.item}>"


class DoubleLinkList(object):
    """双向链表"""

    def __init__(self):
        self._head = None

    def is_empty(self):
        """是否为空"""
        return self._head is None

    def length(self):
        """长度"""
        next_node = self._head
        count = 0
        while next_node is not None:
            count += 1
            next_node = next_node.next
        return count

    def travel(self):
        """遍历"""
        next_node = self._head
        while next_node is not None:
            yield next_node
            next_node = next_node.next

    def add(self, item):
        """头部增加"""
        node = Node(item)
        if not self.is_empty():
            node.next = self._head
            self._head.previous = node
        self._head = node

    def append(self, item):
        """尾部增加"""
        node = Node(item)
        next_node = self._head
        if self.is_empty():
            self._head = node
        else:
            while next_node.next is not None:
                next_node = next_node.next
            next_node.next = node
            node.previous = next_node

    def insert(self, pos, item):
        """指定位置插入"""
        node = Node(item)
        if pos <= 0:
            self.add(item)
        elif pos > self.length() - 1:
            self.append(item)
        else:
            next_node = self._head
            count = 0
            pre_node = next_node
            while next_node is not None:
                count += 1
                pre_node = next_node
                next_node = next_node.next
                if count == pos:
                    break
            if next_node:
                next_node.previous = node
                node.next = next_node
            pre_node.next = node
            node.previous = pre_node

    def remove(self, item):
        """删除节点"""
        next_node = self._head
        while next_node is not None:
            if next_node.item == item:
                if next_node is self._head:
                    self._head = None
                    self._head = next_node.next
                else:
                    next_next_node = next_node.next
                    pre_node = next_node.previous
                    if next_next_node:
                        next_next_node.previous = pre_node
                    if pre_node:
                        pre_node.next = next_next_node
                return
            next_node = next_node.next
        raise IndexError("not found!")

    def search(self, item):
        """查找指定节点"""
        next_node = self._head
        while next_node is not None:
            if next_node.item == item:
                return True
            next_node = next_node.next
        return False

    def __str__(self):
        next_node = self._head
        print_str = ""
        while next_node is not None:
            print_str += f"{next_node.item} <==> "
            next_node = next_node.next
        print_str += "None"
        return print_str


if __name__ == '__main__':
    dll = DoubleLinkList()
    dll.append(1)
    dll.append(2)
    dll.append(7)
    dll.append(6)
    print(dll)
    print(dll.length())
    dll.insert(1, 5)
    print(dll)
    dll.insert(3, 5)
    dll.insert(4, 5)
    print(dll)
    dll.remove(5)
    dll.remove(6)
    dll.remove(1)
    print(dll)
    dll.add(0)
    print(dll)

```

## 1.3 单向循环链表

### 1.3.1 图解

类似单向链表. 不同之处在于最后一个节点的next重新引用到链表头部.

![单向循环链表](image/02-%E9%93%BE%E8%A1%A8/%E5%8D%95%E5%90%91%E5%BE%AA%E7%8E%AF%E9%93%BE%E8%A1%A8.png)

### 1.3.2 实现

```python
# coding = utf-8
class SingleNode(object):
    """Node"""

    def __init__(self, value):
        self.data = value
        self.next = None


# noinspection PyUnboundLocalVariable
class SingleNodeLink(object):
    """link node"""

    def __init__(self):
        self.__head = None
        self.__temp = True
        self.loop_stop = False

    def insert(self, index, value):
        index = index if index >= 0 else len(self) + index + 1
        node = SingleNode(value)
        previous = self.__head
        if not abs(index) <= len(self):
            raise IndexError("index out of range")
        if len(self) == 0:
            self.add(value)
            return
        for i in range(index - 1):
            previous = previous.next
        node.next = previous.next
        previous.next = node

    def append(self, value):
        node = SingleNode(value)
        temp = self.__head
        try:
            while temp.next is not self.__head:
                temp = temp.next
            temp.next = node
            node.next = self.__head
        except AttributeError:
            self.__head = node
            node.next = self.__head

    def __contains__(self, value):
        temp = self.__head
        while temp.next is not self.__head:
            if temp.data == value:
                return True
            temp = temp.next
        else:
            if temp.data == value:
                return True
            return False

    def __len__(self):
        temp = self.__head
        if self.is_empty():
            return 0
        count = 1
        while temp.next is not self.__head:
            temp = temp.next
            count += 1
        return count

    def is_empty(self):
        return self.__head is None

    def add(self, value):
        node = SingleNode(value)
        temp = self.__head
        if temp is None:
            self.__head = node
            node.next = self.__head
            return
        while temp.next is not self.__head:
            temp = temp.next
        node.next = self.__head
        self.__head = node
        temp.next = self.__head

    def __show(self):
        temp = self.__head
        if temp is None:
            raise StopIteration
        while temp.next is not self.__head:
            yield temp.data
            temp = temp.next
        yield temp.data
        raise StopIteration

    def show(self):
        return list(self.__show())

    def __iter__(self):
        return self

    def __next__(self):
        try:
            if self.__temp is True:
                self.__temp = self.__head
            if self.__temp is self.__head and self.loop_stop is True:
                raise AttributeError
            data = self.__temp.data
            self.__temp = self.__temp.next
            self.loop_stop = True
        except AttributeError:
            self.loop_stop = False
            raise StopIteration
        else:
            return data

    def remove(self, value):
        temp = self.__head
        while temp.next is not None:
            if temp.data == value:
                if temp == self.__head:
                    # remove head
                    if len(self) == 1:
                        # only one remove
                        self.__head = None
                        return
                    # remove head
                    tail = self.__head
                    while tail.next is not self.__head:
                        tail = tail.next
                    self.__head = temp.next
                    tail.next = temp.next
                    return
                previous.next = temp.next
                return
            previous = temp
            temp = temp.next
        else:
            if temp.data == value:
                # remove tail
                previous.next = self.__head
                return
            raise ValueError("Not find {!r} value".format(value))


if __name__ == '__main__':
    s = SingleNodeLink()
    print(s.is_empty())
    s.add("aaa")
    s.add("bbb")
    s.add("fff")
    s.append("ccc")
    s.append("ddd")
    s.append("eee")
    print(s.is_empty())
    print(s.show())
    print(len(s))
    # print("eee" in s)
    # print(111 in s)
    print("*****" * 10)
    s.remove("aaa")
    s.remove("bbb")
    s.remove("fff")
    print(s.show())
    # print("*****" * 10)
    s.insert(0, "ccc")
    print(s.show())
    # print("*****" * 10)
    print(tuple(s))
    for i in s:
        print(i)
```
